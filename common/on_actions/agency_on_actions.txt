    on_weekly = { # Operative Slot I Tech

        effect = {
            if = {
                limit = {
                    check_variable = {
                        var:num_equipment@operative_slot_in_warehouse_product > 0
                    }
                }

                set_variable = {
                    var = operative_slot_product
                    value = var:num_equipment@operative_slot_in_warehouse_product
                }
                set_variable = {
                    var = operative_slot_product_negative
                    value = var:num_equipment@operative_slot_in_warehouse_product
                }
                multiply_variable = {
                    var = operative_slot_product_negative
                    value = -1
                }

                ########################################################

                    type = operative_slot_in_warehouse_product
                    amount = operative_slot_product_negative
                }

                add_to_variable = {
                    var = more_production_operative_slot
                    value = operative_slot_product
                }

                set_variable = {
                    var = operative_slot_product
                    value = 0
                }
                set_variable = {
                    var = operative_slot_product_negative
                    value = 0
                }
            }
