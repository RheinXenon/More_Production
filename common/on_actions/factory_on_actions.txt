	on_weekly = { 
		
		effect = {
			if = {
				limit = {
					
					check_variable = {
						var:num_equipment@civilian_factory_in_warehouse_product > 0
					}
				}
				
				set_variable = {
					var = civilian_factory_product
					value = var:num_equipment@civilian_factory_in_warehouse_product
				}
				set_variable = {
					var = civilian_factory_product_negative
					value = var:num_equipment@civilian_factory_in_warehouse_product
				}
				if = {
					limit = {
						check_variable = {
							var:num_equipment@civilian_factory_in_warehouse_product > 0
						}
					}
					multiply_variable = {
						var = civilian_factory_product_negative
						value = -1
					}
				}
				
				
				########################################################

				add_equipment_to_stockpile = {
    				type = civilian_factory_in_warehouse_product
    				amount = civilian_factory_product_negative
    				# producer = ROOT
				}
				add_offsite_building = {
                    type = industrial_complex
                    level = civilian_factory_product
                }
				set_variable = {
					var = civilian_factory_product
					value = 0
				}
				set_variable = {
