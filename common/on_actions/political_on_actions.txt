					limit = {

							var:num_equipment@pp_in_warehouse_product > 0
						}
					}
					multiply_variable = {
						var = political_power_product_negative
						value = -1
					}
				}
				
				
				########################################################

				add_equipment_to_stockpile = {
    				type = pp_in_warehouse_product
    				amount = political_power_product_negative
    				# producer = ROOT
				}

					var = political_power_product
					value = 0
